# Crear proyecto
composer create-project laravel/laravel aulascursosyap

# Crear Interfaces de Pantalla en Laravel/ui
composer require laravel/ui


# Migraciones
## Create 
php artisan make:migration campos_de_usuario
php artisan make:migration create_temas_table
php artisan make:migration create_tipo_actividad_table
php artisan make:migration create_cursos_table
php artisan make:migration create_actividades_table
php artisan make:migration create_roles_de_usuarios_table

## Ejecutar Migracion
php artisan migrate

### Borras las Tablas y la Recrea
php artisan migrate:fresh

### Roallback
php artisan migrate:reset


# Interfaces de autenticacion
php artisan ui bootstrap --auth
npm install
npm run dev

# build ccs sass
npm run build

# Ejecutar el Proyecto
php artisan serve


# Crear los Modelos de las Base de Datos
php artisan make:controller IdiomaController

# Make Controler con los Recursos

# Rutas de las Aplicacion
php artisan route:list

# Rutas de las Aplicacion
 php artisan cache:clear
